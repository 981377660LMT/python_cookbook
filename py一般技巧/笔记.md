9.正则表达式：好用的两个re.findall和re.sub

import re
a = 'python 1111\njava&_678php'

# \d等价于[0-9]
# \D等价于[^0-9]
# \w等价于[A-Za-z0-9_]
# \W等价于[^A-Za-z0-9_]
# \s等价于[ \n\r\t]
# \S等价于[^ \n\r\t]
r = re.findall('\s', a)
print(r)


import re
language = 'PythonC#JavaPHP'
# re.I让正则表达式匹配忽略大小写，多个模式使用|分隔
r = re.findall('c#', language, re.I | re.S)
print(r)

# 默认.不匹配\n，re.S使.能够匹配\n
language = 'PythonC#\nJavaPHP'
r = re.findall('c#.{1}', language, re.I | re.S)
print(r)
